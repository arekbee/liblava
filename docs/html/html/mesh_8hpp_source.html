394template<typename T>
395bool mesh_template<T>::create(device_p d,
396 bool m,
397 VmaMemoryUsage mu) {
398 device = d;
399 mapped = m;
400 memory_usage = mu;
401
402 if (!data.vertices.empty()) {
403 vertex_buffer = make_buffer();
404
405 if (!vertex_buffer->create(device,
406 data.vertices.data(),
407 sizeof(T) * data.vertices.size(),
408 VK_BUFFER_USAGE_VERTEX_BUFFER_BIT,
409 mapped,
410 memory_usage)) {
411 log()->error("create mesh vertex buffer");
412 return false;
413 }
414 }
415
416 if (!data.indices.empty()) {
417 index_buffer = make_buffer();
418
419 if (!index_buffer->create(device,
420 data.indices.data(),
421 sizeof(ui32) * data.indices.size(),
422 VK_BUFFER_USAGE_INDEX_BUFFER_BIT,
423 mapped,
424 memory_usage)) {
425 log()->error("create mesh index buffer");
426 return false;
427 }
428 }
429
430 return true;
431}

500//-----------------------------------------------------------------------------
501template<typename T,
502 bool generate_colors,
503 bool generate_normals,
504 bool generate_uvs,
505 bool has_colors,
506 bool has_normals,
507 bool has_uvs>
508mesh_template_data<T> create_mesh_data(mesh_type type) {
509 mesh_template_data<T> return_mesh_data;
510
511 constexpr bool auto_position = requires(const T t) {
512 t.position;
513 };
514 static_assert(auto_position,
515 "Vertex struct `T` must contain field `position`");
516
517 constexpr bool auto_colors = requires(const T t) {
518 t.color;
519 };
520 constexpr bool auto_normals = requires(const T t) {
521 t.normal;
522 };
523 constexpr bool auto_uvs = requires(const T t) {
524 t.uv;
525 };
526
527 using PosType = decltype(T::position);
528
529 switch (type) {
530 case mesh_type::cube: {
531 return_mesh_data.indices.reserve(36);
532 return_mesh_data.indices = make_primitive_indices_cube<(generate_normals
533 && auto_normals)>();
534 constexpr size_t vert_count = (generate_normals && auto_normals) ? 24 : 8;
535 return_mesh_data.vertices.reserve(vert_count);
536 auto positions = make_primitive_positions_cube<PosType, vert_count,
537 (generate_normals && auto_normals)>();
538 for (size_t i = 0; i < vert_count; i++) {
539 T vert;
540 vert.position = positions[i];
541 if constexpr (generate_normals && auto_normals) {
542 // this array is generated inside of every loop because
543 // that makes the scoping rules simplest to follow
544 // my expectation is that a compiler should be able
545 // to trivially optimize this
546 using NormType = decltype(T::normal);
547 auto normals = make_primitive_normals_cube<NormType>();
548 vert.normal = normals[i / 4];
549 }
550 if constexpr (generate_uvs && auto_uvs) {
551 using UVType = decltype(T::uv);
552 auto uvs = make_primitive_uvs_cube<UVType>();
553 vert.uv = uvs[i];
554 }
555 return_mesh_data.vertices.push_back(vert);
556 }
557 break;
558 }
559
560 case mesh_type::triangle: {
561 return_mesh_data.vertices.reserve(3);
562 T vert_one;
563 vert_one.position = { 1, 1, 0 };
564 T vert_two;
565 vert_two.position = { -1, 1, 0 };
566 T vert_three;
567 vert_three.position = { 0, -1, 0 };
568 if constexpr (generate_uvs && auto_uvs) {
569 vert_one.uv = { 1, 1 };
570 vert_two.uv = { 0, 1 };
571 vert_three.uv = { 0.5, 0 };
572 }
573 if constexpr (generate_normals && auto_normals) {
574 vert_one.normal = { 1, 1, 0 };
575 vert_two.normal = { -1, 1, 0 };
576 vert_three.normal = { 0, -1, 0 };
577 }
578 return_mesh_data.vertices.push_back(vert_one);
579 return_mesh_data.vertices.push_back(vert_two);
580 return_mesh_data.vertices.push_back(vert_three);
581 break;
582 }
583
584 case mesh_type::quad: {
585 return_mesh_data.vertices.reserve(4);
586 return_mesh_data.indices.reserve(6);
587 T vert_one;
588 vert_one.position = { 1, 1, 0 };
589 T vert_two;
590 vert_two.position = { -1, 1, 0 };
591 T vert_three;
592 vert_three.position = { -1, -1, 0 };
593 T vert_four;
594 vert_four.position = { 1, -1, 0 };
595 if constexpr (generate_uvs && auto_uvs) {
596 vert_one.uv = { 1, 1 };
597 vert_two.uv = { 0, 1 };
598 vert_three.uv = { 0, 0 };
599 vert_four.uv = { 1, 0 };
600 }
601 if constexpr (generate_normals && auto_normals) {
602 vert_one.normal = { 0, 0, 1 };
603 vert_two.normal = { 0, 0, 1 };
604 vert_three.normal = { 0, 0, 1 };
605 vert_four.normal = { 0, 0, 1 };
606 }
607 // clang-format off
608 return_mesh_data.indices = {
609 0, 1, 2,
610 2, 3, 0,
611 };
612 // clang-format on
613 return_mesh_data.vertices.push_back(vert_one);
614 return_mesh_data.vertices.push_back(vert_two);
615 return_mesh_data.vertices.push_back(vert_three);
616 return_mesh_data.vertices.push_back(vert_four);
617 break;
618 }
619
620 case mesh_type::hexagon: {
621 return_mesh_data.vertices.reserve(7);
622 return_mesh_data.indices.reserve(18);
623 hex_layout layout;
624 layout.orientation = hex_layout_point_y;
625 layout.size = { 1, 1 };
626 auto hex_corners = hex_polygon_corners(layout, {});
627 T vert_center;
628 vert_center.position = { 0, 0, 0 };
629 T vert_sw;
630 PosType temp;
631 temp[0] = hex_corners.at(0).x;
632 temp[1] = hex_corners.at(0).y;
633 temp[2] = 0;
634 vert_sw.position = temp;
635 T vert_nw;
636 temp[0] = hex_corners.at(1).x;
637 temp[1] = hex_corners.at(1).y;
638 vert_nw.position = temp;
639 T vert_n;
640 temp[0] = hex_corners.at(2).x;
641 temp[1] = hex_corners.at(2).y;
642 vert_n.position = temp;
643 T vert_ne;
644 temp[0] = hex_corners.at(3).x;
645 temp[1] = hex_corners.at(3).y;
646 vert_ne.position = temp;
647 T vert_se;
648 temp[0] = hex_corners.at(4).x;
649 temp[1] = hex_corners.at(4).y;
650 vert_se.position = temp;
651 T vert_s;
652 temp[0] = hex_corners.at(5).x;
653 temp[1] = hex_corners.at(5).y;
654 vert_s.position = temp;
655 if constexpr (generate_uvs && auto_uvs) {
656 vert_center.uv = { 0, 0 };
657 vert_sw.uv = { 1, 0 };
658 vert_nw.uv = { 0, 1 };
659 vert_n.uv = { 1, 1 };
660 vert_ne.uv = { 1, 0 };
661 vert_se.uv = { 0, 1 };
662 vert_s.uv = { 1, 1 };
663 }
664 if constexpr (generate_normals && auto_normals) {
665 vert_center.normal = { 0, 0, 1 };
666 vert_sw.normal = { 0, 0, 1 };
667 vert_nw.normal = { 0, 0, 1 };
668 vert_n.normal = { 0, 0, 1 };
669 vert_ne.normal = { 0, 0, 1 };
670 vert_se.normal = { 0, 0, 1 };
671 vert_s.normal = { 0, 0, 1 };
672 }
673 // clang-format off
674 return_mesh_data.indices = {
675 0, 1, 6, 0, 6, 5, 0, 5, 4, 0, 4, 3, 0, 3, 2, 0, 2, 1
676 };
677 // clang-format on
678 return_mesh_data.vertices.push_back(vert_center);
679 return_mesh_data.vertices.push_back(vert_sw);
680 return_mesh_data.vertices.push_back(vert_nw);
681 return_mesh_data.vertices.push_back(vert_n);
682 return_mesh_data.vertices.push_back(vert_ne);
683 return_mesh_data.vertices.push_back(vert_se);
684 return_mesh_data.vertices.push_back(vert_s);
685 break;
686 }
687
688 case mesh_type::none:
689 default:
690 break;
691 }
692
693 if constexpr (generate_colors && auto_colors) {
694 for (auto& vert : return_mesh_data.vertices) {
695 // this does not work on glm vectors
696 // for (auto& this_color : vert.color) {
697 // for (auto& color_component : this_color) {
698 // color_component = 1;
699 // }
700 // }
701 if constexpr (std::is_same<decltype(vert.color), glm::vec3>::value) {
702 vert.color = { 1, 1, 1 };
703 } else if constexpr (std::is_same<decltype(vert.color), glm::vec4>::value) {
704 vert.color = { 1, 1, 1, 1 };
705 } else {
706 for (size_t i = 0; i < vert.color.size(); i++) {
707 vert.color[i] = 1;
708 }
709 }
710 }
711 }
712
713 return return_mesh_data;
714}

336inline hex_cell hex_round(hex_frac_cell const& cell) {
337 auto qi = to_i32(std::round(cell.q));
338 auto ri = to_i32(std::round(cell.r));
339 auto si = to_i32(std::round(cell.s));
340
341 auto q_diff = std::abs(qi - cell.q);
342 auto r_diff = std::abs(ri - cell.r);
343 auto s_diff = std::abs(si - cell.s);
344
345 if ((q_diff > r_diff) && (q_diff > s_diff))
346 qi = -ri - si;
347 else if (r_diff > s_diff)
348 ri = -qi - si;
349 else
350 si = -qi - ri;
351
352 return { qi, ri, si };
353}

382inline hex_cell::list hex_line(hex_cell const& a,
383 hex_cell const& b) {
384 auto a_nudge = hex_frac_cell{ a.q + 0.000001f,
385 a.r + 0.000001f,
386 a.s - 0.000002f };
387 auto b_nudge = hex_frac_cell{ b.q + 0.000001f,
388 b.r + 0.000001f,
389 b.s - 0.000002f };
390
391 ui32 n = hex_distance(a, b);
392 auto step = 1.0 / std::max(n, 1u);
393
394 hex_cell::list results;
395 for (auto i = 0u; i <= n; ++i)
396 results.push_back(hex_round(hex_lerp(a_nudge,
397 b_nudge,
398 step * i)));
399
400 return results;
401}

565inline hex_point hex_to_pixel(hex_layout const& layout,
566 hex_cell const& cell) {

568 auto size = layout.size;
569 auto origin = layout.origin;

583inline hex_frac_cell hex_pixel_to_cell(hex_layout const& layout,
584 hex_point const& p) {
585 auto m = layout.orientation;
586 auto size = layout.size;
587 auto origin = layout.origin;
588 auto pt = hex_point{ (p.x - origin.x) / size.x,
589 (p.y - origin.y) / size.y };
590 auto q = m.b0 * pt.x + m.b1 * pt.y;
591 auto r = m.b2 * pt.x + m.b3 * pt.y;
592 return { q, r, -q - r };
